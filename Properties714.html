Database Schema
PostgreSQL — 5 tables · Ready for n8n integration
users
CREATE TABLE users (
  id SERIAL PRIMARY KEY,
  name TEXT,
  email TEXT,
  investment_focus TEXT,
  risk_tolerance TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);
properties
CREATE TABLE properties (
  id SERIAL PRIMARY KEY,
  address TEXT NOT NULL,
  city TEXT,
  state TEXT,
  zip TEXT,
  asking_price NUMERIC,
  estimated_arv NUMERIC,
  estimated_rehab NUMERIC,
  strategy_candidate TEXT,
  status TEXT DEFAULT 'new',
  created_at TIMESTAMP DEFAULT NOW()
);
comparables
CREATE TABLE comparables (
  id SERIAL PRIMARY KEY,
  property_id INTEGER REFERENCES properties(id),
  address TEXT,
  sold_price NUMERIC,
  sqft INTEGER,
  beds INTEGER,
  baths NUMERIC,
  distance NUMERIC,
  sold_date DATE,
  source TEXT
);
deal_analysis
CREATE TABLE deal_analysis (
  id SERIAL PRIMARY KEY,
  property_id INTEGER REFERENCES properties(id),
  arv NUMERIC,
  purchase_price NUMERIC,
  rehab_cost NUMERIC,
  closing_costs NUMERIC,
  holding_cost NUMERIC,
  agent_fees NUMERIC,
  total_investment NUMERIC,
  estimated_profit NUMERIC,
  roi NUMERIC,
  deal_score INTEGER,
  recommendation TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);
strategies
CREATE TABLE strategies (
  id SERIAL PRIMARY KEY,
  property_id INTEGER REFERENCES properties(id),
  strategy_type TEXT,        -- fix_flip | rental | wholesale
  estimated_rent NUMERIC,
  monthly_cashflow NUMERIC,
  cap_rate NUMERIC,
  cash_on_cash NUMERIC,
  assignment_fee NUMERIC,
  created_at TIMESTAMP DEFAULT NOW()
);
